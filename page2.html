We update the posterior for each die, to be the new prior. (calculate 3 posteriors, in this instance, since the first two die have probability zero).

The prior is different than the given distribution, and will update every time. Eventually, you will get a more accurate probability distribution. We will maintain our previous information by multiplying by previous prior distributions in the numerator.

import matplotlib.pyplot as plt
from __future__ import division



class Bayes(object):
    '''
    INPUT:
        prior (dict): key is the value (e.g. 4-sided die),
                      value is the probability

        likelihood_func (function):
                        INPUT: INT data--the value of the roll of a die 
                               INT key--the maximum value of the die (ie pass
                                            4 to get the likelihood for a
                                            4-sided die)
    '''
    def __init__(self, prior, likelihood_func):
        self.prior = prior.copy()
        self.likelihood_func = likelihood_func


    def _normalize(self):
        '''
        INPUT: None
        OUTPUT: None

        Makes the sum of the probabilities equal 1.
        '''
        self.prior = {k: p/sum(self.prior.itervalues()) for k, p in self.prior.iteritems()}
            
            
    def update(self, data):
        '''
        INPUT:
            data (int or str): A single observation (data point)

        OUTPUT: None

        Conduct a bayesian update. Multiply the prior by the likelihood and
        make this the new prior.
        '''

        self.prior = {die: round(self.likelihood_func(data, die) * p, 3) for die, p in self.prior.copy().iteritems()}
        self._normalize()


    def print_distribution(self):
        '''
        Print the current posterior probability.
        '''
        print [(key, round(self.prior[key], 3)) for key in sorted(self.prior)]

    def plot(self, color=None, title=None, label=None):
        '''
        Plot the current prior.
        '''
        plt.scatter(self.prior.keys(),self.prior.values(), alpha=0.5)
    
    
def likelihood(roll, die):
    if roll in range(1, die+1):
        return 1 / die
    else:
        return 0
